public class MaxValueFinder {  
  
    /**  
     * 返回给定数组中的最大值。  
     *  
     * @param list 输入的整数数组  
     * @return 数组中的最大值  
     * @throws IllegalArgumentException 如果输入的数组为空  
     */  
    public int largest(int[] list) {  
        // 检查数组是否为空  
        if (list == null || list.length == 0) {  
            throw new IllegalArgumentException("数组不能为空");  
        }  
  
        // 初始化最大值为数组的第一个元素  
        int max = list[0];  
  
        // 遍历数组，更新最大值  
        for (int i = 1; i < list.length; i++) {  
            if (list[i] > max) {  
                max = list[i];  
            }  
        }  
  
        // 返回最大值  
        return max;  
    }  
}
